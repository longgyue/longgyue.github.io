Python编写指定路径文件的内容替换自定义的内容-正则表达式读写文件的应用

#在自定义位置替换自定义文字
import os,sys,pyperclip,re

filename=sys.argv[1]#获取操作的文件位置
print(filename)#打印文件路径信息

file=open(filename,‘r’)#只读方式打开文件
txtdata=file.read()#读取文件

file.close()#关闭文件

print(txtdata)#打印文件内容
print (‘where do you want to replace ? please input the word:’)
place=input()#输入替换的位置
print(‘input the wanted word :’)
indata=input()#输入要替换的文字
nameRegex=re.compile(r‘’+str(place))#查找相关文字
newstring=nameRegex.sub(indata,txtdata)#替换文字
print(newstring)#打印文字
file=open(filename,‘w’)#以写的方式打开文字
file.write(newstring)#写入新数据
file.close()#关闭文件
